//Use esse import para pegar as funções necessarias.
import { useState } from "react";

//Essa parte lança para a interface e para o console dentro do React
export default function App(){
  const [input, setInput] = useState("")
  const [idade, setIdade] = useState("")

//Funcao do botão.
  function mostraAluno(){
    setAluno(input)
    setIdade(idade)
    console.log(idade)
    
  }

  return(

    <div>
     <h1>Conhecendo useState</h1>

     //Aqui que faz a Solicitação dos dados
     <input
        placeholder='Digite o Nome:'
        value={input}
        onChange={ (e) => setInput(e.target.value)}
     />

     <br/>

     <input
        placeholder="Digite sua idade"
        value={idade}
        onChange={ (e) => setIdade(e.target.value)}
     />

    //quebra uma linha
     <br/>
     <br/>

    //Adiciona um botão na interface
     <button onClick={mostraAluno}>Mostrar Aluno</button>

     <hr/>

     <h3>Bem vindo: {aluno} idade: {idade}</h3>

    </div>
  )
}
